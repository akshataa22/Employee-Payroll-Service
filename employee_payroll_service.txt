CREATE DATABASE payroll_service;
SHOW DATABASES;
USE payroll_service;

CREATE TABLE employee_payroll (
    id INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    name VARCHAR(50) NOT NULL,
    salary DOUBLE NOT NULL,
    start_date DATE NOT NULL
);

SELECT * FROM employee_payroll;

SELECT salary FROM employee_payroll
WHERE name = 'John doe';
SELECT * FROM employee_payroll
WHERE start_date BETWEEN CAST('2020-03-15' AS DATE) AND DATE(NOW());

ALTER TABLE employee_payroll ADD gender CHAR(1) AFTER name;
UPDATE employee_payroll SET gender = 'M' WHERE name = 'John Doe';
UPDATE employee_payroll SET gender = 'F' WHERE name = 'Alice Jobs';
UPDATE employee_payroll SET gender = 'M' WHERE name = 'Peter Davidson';
